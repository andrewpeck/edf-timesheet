  return functools.reduce(lambda x, y:x + '''", "''' + y,a)

  'L0MDT': ("L0MDT",),
  'ETL': ("ETL",),
  'GEX1': ("GE11", "GE21"),
  'GE21': ("GE21",),
  'CSC': ("CSC",),
