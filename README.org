** Task Distribution

#+begin_src bash :results output :exports results
cat wordcloud.txt \
| tr -d '[:punct:]' \
| awk 'length($0)>3' \
| grep "\S" | sort | uniq -c \
| awk 'BEGIN { FS = " " } ; { printf("%s, %s\n", $2, $1) }' \
| sort -r -n -t ',' -k 2 \
| grep -v "with" \
| grep -v "work" \
| head -n 25 \
| uplot bar -d, -t "Tasks" 2>&1
