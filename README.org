** Task Distribution

#+begin_src bash :results output :exports results
cat wordcloud.txt \
| tr -d '[:punct:]' \
| awk 'length($0)>3' \
| grep "\S" | sort | uniq -c \
| awk 'BEGIN { FS = " " } ; $1 < 40 { misc += $1; next} { printf("%s, %s\n", $2, $1) }; END {print "misc, " misc}' \
| sort -r -n -t ',' -k 2 \
| grep -v "with" \
| grep -v "work" \
| uplot bar -d, -t "Tasks" 2>&1
